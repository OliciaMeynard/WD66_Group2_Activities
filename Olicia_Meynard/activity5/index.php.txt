<?php
	
	$item1 = 100;
	$item2 = 200;
	$item3 = 300;
	
	
	function computeStocksQty ($itm, $qty) {
	  
	    global $item1, $item2, $item3;
	    
        if($qty <= 0){
            return "Invalid Quantity".'<br>';
        }
        
	    switch($itm){
	      case 1:
	        $item1 -= $qty;
	        if($item1 <= 0){
	          return "No more Stocks for Item $itm".'<br>';
	        }
	        return "Item $itm- ($qty) qty | Remaining Stocks = $item1".'<br>';
	        break;
	     case 2:
	        $item2 -= $qty;
	        if($item2 <= 0){
	          return "No more Stocks for Item $itm".'<br>';
	        }
	        return "Item $itm- ($qty) qty  | Remaining Stocks = $item2".'<br>';
	        break;

        case 3:
            $item3 -= $qty;
            if($item3 <= 0){
            return "No more Stocks for Item $itm".'<br>';
            }
            return "Item $itm - ($qty) qty  | Remaining Stocks = $item3".'<br>';
            break;


	        default:
	        return "Invalid input: Item $itm".'<br>';
	    }

  
	      
	 
	}
	 
	echo "Given Quantities on example".'<br>';
    echo "\n";
	echo computeStocksQty(1, 3 );
    echo "\n";
	echo computeStocksQty(1, 5 );
    echo "\n";
	echo computeStocksQty(2, 5 );
    echo "\n";
	echo computeStocksQty(1, 50 );
    echo "\n";
	echo computeStocksQty(3, 5 );
    echo "\n";
    echo "\n";
    echo "\n";

	

    echo "Sample quantities from Invalid inputs to No stocks".'<br>'; 
    echo "\n";

    echo "\n";

    echo "\n";

    echo "\n";
	echo computeStocksQty(2, 200 );
    echo "\n";
